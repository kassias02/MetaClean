previewSize.textContent = formatBytes(fileSize); // Afficher le résumé des métadonnées metadataSummary.innerHTML = ''; // Compter les métadonnées et mettre à jour le compteur const metadataItems = countMetadataItems(metadata); metadataCount.textContent = `${metadataItems} ${getTranslation('metadata_items')}`; // Créer un résumé des métadonnées importantes const importantMetadata = extractImportantMetadata(metadata); if (Object.keys(importantMetadata).length === 0) { const noMetadata = document.createElement('p'); noMetadata.className = 'text-gray-600'; noMetadata.textContent = getTranslation('no_metadata'); metadataSummary.appendChild(noMetadata); } else { // Afficher les métadonnées importantes for (const [category, value] of Object.entries(importantMetadata)) { const metadataItem = document.createElement('div'); metadataItem.className = 'flex items-center justify-between py-2 border-b border-gray-100'; const label = document.createElement('span'); label.className = 'font-medium'; label.textContent = getTranslation(`metadata_${category.toLowerCase()}`); const valueEl = document.createElement('span'); valueEl.className = 'text-gray-600'; valueEl.textContent = value; // Ajouter un tag pour les informations sensibles if (['GPS', 'Date', 'Device'].includes(category)) { const tag = document.createElement('span'); tag.className = 'metadata-tag metadata-warning text-xs ml-2'; tag.textContent = 'Sensible'; label.appendChild(tag); } metadataItem.appendChild(label); metadataItem.appendChild(valueEl); metadataSummary.appendChild(metadataItem); } } // Remplir les métadonnées détaillées displayAllMetadata(metadata, allMetadataContent); } // Extraire les métadonnées importantes pour l'aperçu function extractImportantMetadata(metadata) { const important = {}; if (!metadata || !metadata.exif) { return important; } const exif = metadata.exif; // GPS if (exif.GPSLatitude && exif.GPSLongitude) { const lat = convertDMSToDD(exif.GPSLatitude, exif.GPSLatitudeRef); const lng = convertDMSToDD(exif.GPSLongitude, exif.GPSLongitudeRef); important['GPS'] = `${lat.toFixed(6)}, ${lng.toFixed(6)}`; } // Date if (exif.DateTime) { important['Date'] = exif.DateTime; } else if (exif.DateTimeOriginal) { important['Date'] = exif.DateTimeOriginal; } else if (exif.DateTimeDigitized) { important['Date'] = exif.DateTimeDigitized; } // Appareil if (exif.Make && exif.Model) { important['Device'] = `${exif.Make} ${exif.Model}`; } else if (exif.Make) { important['Device'] = exif.Make; } else if (exif.Model) { important['Device'] = exif.Model; } // Logiciel if (exif.Software) { important['Software'] = exif.Software; } // Auteur/Copyright if (exif.Artist) { important['Author'] = exif.Artist; } if (exif.Copyright) { important['Copyright'] = exif.Copyright; } // Résolution if (exif.ImageWidth && exif.ImageHeight) { important['Resolution'] = `${exif.ImageWidth} × ${exif.ImageHeight}`; } else if (exif.PixelXDimension && exif.PixelYDimension) { important['Resolution'] = `${exif.PixelXDimension} × ${exif.PixelYDimension}`; if (ref === 'S' || ref === 'W') { dd = dd * -1; } return dd; } // Formater le temps d'exposition function formatExposureTime(time) { if (time < 1) { return `1/${Math.round(1/time)}`; } else { return time.toString(); } } // Afficher toutes les métadonnées function displayAllMetadata(metadata, container) { container.innerHTML = ''; if (!metadata || !metadata.exif || Object.keys(metadata.exif).length === 0) { const noMetadata = document.createElement('p'); noMetadata.className = 'text-gray-600'; noMetadata.textContent = getTranslation('no_metadata'); container.appendChild(noMetadata); return; } const table = document.createElement('table'); table.className = 'w-full text-sm'; keyCell.textContent = formatExifKey(key); row.appendChild(keyCell); const valueCell = document.createElement('td'); valueCell.className = 'py-1 text-gray-600'; valueCell.textContent = formatExifValue(value); row.appendChild(valueCell); table.appendChild(row); } } container.appendChild(table); } // Formater la clé EXIF function formatExifKey(key) { return key.replace(/([A-Z])/g, ' $1').replace(/^./, str => str.toUpperCase()); } // Formater la valeur EXIF function formatExifValue(value) { if (value instanceof Array) { return value.join(', '); } return value.toString(); } // Formater la taille du fichier function formatBytes(bytes, decimals = 2) { if (bytes === 0) return '0 Bytes'; const k = 1024; const dm = decimals < 0 ? 0 : decimals; const sizes = ['Bytes', 'KB', 'MB', 'GB', 'TB']; const i = Math.floor(Math.log(bytes) / Math.log(k)); return parseFloat((bytes / Math.pow(k, i)).toFixed(dm)) + ' ' + sizes[i]; } // Obtenir une traduction function getTranslation(key) { const lang = getCurrentLanguage(); if (translations[lang] && translations[lang][key]) { return translations[lang][key]; } return key; } // Obtenir la langue actuelle function getCurrentLanguage() { const currentLang = document.getElementById('current-language').textContent; for (const option of document.querySelectorAll('.language-option')) { if (option.textContent === currentLang) { return option.getAttribute('data-lang'); } } return 'fr'; } // Nettoyer l'image (supprimer les métadonnées) function createCleanImage(originalDataUrl, mimeType, callback) { const img = new Image(); // Estimer la taille du fichier nettoyé const cleanSize = Math.round(cleanDataUrl.length * 0.75); callback(cleanDataUrl, cleanSize); }; img.onerror = function() { console.error("Erreur lors du chargement de l'image dans Canvas"); alert("Une erreur s'est produite lors du traitement de l'image."); }; img.src = originalDataUrl; } // Réinitialiser l'interface utilisateur function resetUI() { dropZone.classList.remove('hidden'); processingArea.classList.add('hidden'); metadataPreviewArea.classList.add('hidden'); resultsArea.classList.add('hidden'); metadataDetails.classList.add('hidden'); allMetadataDetails.classList.add('hidden'); progressBar.style.width = '0%'; fileInput.value = ''; // Réinitialiser les variables globales currentFile = null; currentImageData = null; currentMetadata = null; } // Événements pour les boutons // Afficher toutes les métadonnées showAllMetadataBtn.addEventListener('click', function() { allMetadataDetails.classList.toggle('hidden'); showAllMetadataBtn.textContent = allMetadataDetails.classList.contains('hidden') ? getTranslation('show_all_metadata') : getTranslation('hide_all_metadata'); }); // Supprimer les métadonnées removeMetadataBtn.addEventListener('click', function() { if (!currentImageData) { return; } // Afficher le traitement metadataPreviewArea.classList.add('hidden'); processingArea.classList.remove('hidden'); progressBar.style.width = '50%'; statusText.textContent = getTranslation('removing_metadata'); // Supprimer les métadonnées setTimeout(() => { createCleanImage(currentImageData, currentFile.type, (cleanImageUrl, cleanSize) => { // Afficher les résultats originalImage.src = currentImageData; cleanedImage.src = cleanImageUrl; originalSize.textContent = formatBytes(currentFile.size); cleanedSize.textContent = formatBytes(cleanSize); downloadBtn.href = cleanImageUrl; downloadBtn.download = getCleanFileName(currentFile.name); // Afficher les métadonnées supprimées displayAllMetadata(currentMetadata, metadataContent); // Mettre à jour l'interface progressBar.style.width = '100%'; setTimeout(() => { processingArea.classList.add('hidden'); resultsArea.classList.remove('hidden'); }, 500); }); }, 800); }); // Afficher/masquer les métadonnées dans les résultats viewMetadataBtn.addEventListener('click', function() { metadataDetails.classList.toggle('hidden'); viewMetadataBtn.textContent = metadataDetails.classList.contains('hidden') ? getTranslation('view_metadata') : getTranslation('hide_metadata'); }); // Traiter une autre image processAnotherBtn.addEventListener('click', function() { resetUI(); }); // Générer un nom de fichier propre function getCleanFileName(originalName) { const baseName = originalName.substring(0, originalName.lastIndexOf('.')) || originalName; const extension = originalName.split('.').pop(); À propos de MetaClean

MetaClean a été créé pour aider les utilisateurs à protéger leur vie privée en ligne. Les métadonnées des photos peuvent révéler de nombreuses informations sensibles comme votre localisation, le modèle de votre appareil, la date et l'heure exactes, et même parfois des informations personnelles.

Notre mission est de rendre accessible à tous un outil simple, gratuit et sécurisé pour nettoyer ces données avant de partager vos images en ligne.

Nous croyons fermement que la confidentialité en ligne est un droit fondamental, et nous nous engageons à fournir des outils qui respectent ce principe.
